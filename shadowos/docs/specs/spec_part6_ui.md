# File: userspace/shadow-control-center/tabs/defense.py

class DefenseTab(Gtk.Box):
    def __init__(self):
        super().__init__(orientation=Gtk.Orientation.VERTICAL, spacing=10)
        
        # Scan Detection Section
        scan_frame = Gtk.Frame(label="Scan Detection")
        scan_box = Gtk.Box(orientation=Gtk.Orientation.VERTICAL, spacing=5)
        
        # Enable toggle
        self.scan_enable = Gtk.Switch()
        self.scan_enable.connect("state-set", self.on_scan_toggle)
        
        # Sensitivity slider
        self.sensitivity = Gtk.Scale.new_with_range(
            Gtk.Orientation.HORIZONTAL, 1, 10, 1
        )
        self.sensitivity.set_value(5)
        
        # Action radio buttons
        self.action_alert = Gtk.RadioButton.new_with_label(None, "Alert Only")
        self.action_divert = Gtk.RadioButton.new_with_label_from_widget(
            self.action_alert, "Alert + Divert"
        )
        self.action_block = Gtk.RadioButton.new_with_label_from_widget(
            self.action_alert, "Alert + Block"
        )
        
        # ... assemble widgets ...
        
    def on_scan_toggle(self, switch, state):
        shadow.set_enabled("detect", state)
